Step 7: Create a for loop function to explore Ks from 2 to 10 for each method using the Silhouette score and display the results with a plot.

Ward’s Method

morphoward_2 <- numeric(10)

for(k in 2:10) {
  morphoward_clusters <- cutree(morpho_ward, k)
  morphoward_sil <- silhouette(morphoward_clusters, d)
  morphoward_2[k] <- mean(morphoward_sil[, 3])
}

plot(1:10, morphoward_2, type="b", pch=19, xlab="Number of Clusters", ylab="Silhouette Width", main="Silhouette Scores for Ward Method Clustering")
